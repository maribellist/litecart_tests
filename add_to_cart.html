add_to_cart
open /en/
storeEval Math.floor(1 + Math.random()*9) amount
click css=li.product a[title='Rubber Duck']
waitForElementPresent name=quantity
storeText css=h1.title title
storeElementPresent css=div#box-product strong.campaign-price has_discount
if ${has_discount}
storeText css=div#box-product strong.campaign-price price
else
storeText css=div#box-product span.price price
endIf
storeEval "${price}".substring(1) price
storeEval ${amount}*${price} total
type name=quantity ${amount}
storeElementPresent name=options[Size] has_size
if ${has_size}
select name=options[Size] label=Small
endIf
click name=add_cart_product
waitForText css=div#cart a.content Cart:
${amount} item(s) - $${total}
click css=div#cart a.content
waitForElementPresent css=div#checkout-customer-wrapper
assertAttribute name=quantity@value ${amount}
assertText //div[@id='box-checkout-cart']/div/ul/li/form/div/p/a/strong ${title}
